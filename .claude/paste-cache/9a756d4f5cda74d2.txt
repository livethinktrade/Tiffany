Security-Hardened Setup

 ### Step 1: Install autossh

 ```bash
   sudo apt update && sudo apt install autossh -y
 ```

 ### Step 2: Create Systemd Service (Security Hardened)

 ```bash
   sudo tee /etc/systemd/system/clawdbot-webhook-tunnel.service > /dev/null << 'EOF'
   [Unit]
   Description=Clawdbot Webhook SSH Tunnel
   After=network.target

   [Service]
   Type=simple
   User=root
   ExecStart=/usr/bin/autossh -M 0 \
       -o StrictHostKeyChecking=no \
       -o UserKnownHostsFile=/dev/null \
       -o ServerAliveInterval=60 \
       -o ServerAliveCountMax=3 \
       -o BatchMode=yes \
       -N \
       -R 80:localhost:18789 \
       nokey@localhost.run
   Restart=always
   RestartSec=10
   NoNewPrivileges=true
   ProtectSystem=strict
   ProtectHome=true
   ReadWritePaths=/tmp

   [Install]
   WantedBy=multi-user.target
   EOF
 ```

 ### Step 3: Enable and Start

 ```bash
   sudo systemctl daemon-reload
   sudo systemctl enable clawdbot-webhook-tunnel
   sudo systemctl start clawdbot-webhook-tunnel
   sudo systemctl status clawdbot-webhook-tunnel
 ```
